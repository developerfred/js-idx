Interfaces and types

CeramicApi#

Ceramic API interface exported by the @ceramicnetwork/ceramic-common library

DID#

DID instance exported by the dids library

DIDProvider#

DID Provider interface exported by the dids library

Doctype#

Doctype interface exported by the @ceramicnetwork/ceramic-common library

JWSSignature#

JWS Signature interface exported by the dids library

Resolver#

Resolver instance exported by the did-resolver library

ResolverOptions#

ResolverOptions interface exported by the dids library

DocID#

The ID of a Ceramic Document

type DocID = string

Schema#

A type representing a JSON Schema

type Schema = Record<string, unknown>

Definition#

A Definition is a Ceramic Document describing a resource in the Identity Index

interface Definition<T = unknown> {
name: string
schema: DocID
description?: string
url?: string
config?: T
}

DefinitionsAliases#

type DefinitionsAliases = Record<string, DocID>

Entry#

An entry in the Identity Index

interface Entry {
tags: Array<string>
ref: DocID
}

DefinitionEntry#

interface DefinitionEntry extends Entry {
def: DocID
}

ContentEntry#

interface ContentEntry extends DefinitionEntry {
content: unknown
}

IdentityIndexContent#

Represents the shape of the content stored in the Identity Index

type IdentityIndexContent = Record<DocID, Entry>

EncodedDagJWS#

interface EncodedDagJWS {
payload: string
signatures: Array<JWSSignature>
link: string
}

EncodedDagJWSResult#

interface EncodedDagJWSResult {
jws: EncodedDagJWS
linkedBlock: string
}

IDXDefinitionName#

Name aliases of Definitions provided by the IDX tools library

type IDXDefinitionName = 'basicProfile' | 'cryptoAccountLinks'

IDXSignedDefinitions#

Signed Definitions provided by the IDX tools library

type IDXSignedDefinitions = Record<IDXDefinitionName, DagJWSResult>

IDXPublishedDefinitions#

Record of Definitions published to Ceramic

type IDXPublishedDefinitions = Record<IDXDefinitionName, DocID>

IDXSchemaName#

Names of Schemas provided by the IDX tools library

type IDXSchemaName = 'BasicProfile' | 'CryptoAccountLinks' | 'Definition' | 'IdentityIndex'

IDXSignedSchemas#

Signed Schemas provided by the IDX tools library

type IDXSignedSchemas = Record<IDXSchemaName, DagJWSResult>

IDXPublishedSchemas#

Record of Schemas published to Ceramic

type IDXPublishedSchemas = Record<IDXSchemaName, DocID>

IDXPublishedConfig#

Definitions and Schemas published to Ceramic

interface IDXPublishedConfig {
definitions: IDXPublishedDefinitions
schemas: IDXPublishedSchemas
}

PublishDoc#

interface PublishDoc<T = unknown> {
id?: DocID
content: T
owners?: Array<string>
schema?: DocID
}

DefinitionDoc#

interface DefinitionDoc extends PublishDoc<Definition> {
id: DocID
}

SchemaDoc#

interface SchemaDoc extends PublishDoc<Schema> {
name: string
}

IDXOptions#

Options used by the IDX class constructor

interface IDXOptions {
autopin?: boolean
ceramic: CeramicApi
definitions?: DefinitionsAliases
resolver?: ResolverOptions
schemas: SchemasAliases
}

AuthenticateOptions#

Options used by the authenticate method of the IDX class

interface AuthenticateOptions {
paths?: Array<string>
provider?: DIDProvider
}

CreateOptions#

Options used by the createDocument method of the IDX class

interface CreateOptions {
pin?: boolean
}

CreateContentOptions#

Options used by the setEntryContent method of the IDX class

interface CreateContentOptions extends CreateOptions {
tags?: Array<string>
}

ContentIteratorOptions#

Options used by the contentIterator method of the IDX class

interface ContentIteratorOptions {
did?: string
tag?: string
}
